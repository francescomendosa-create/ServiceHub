    match /artifacts/{appId}/publicDigital/{shareId} {
      // Chiunque con il link può LEGGERE lo snapshot (solo lettura sul tablet / PC remoto).
      allow read: if true;
      // Qualsiasi dispositivo autenticato del team può creare/aggiornare lo stesso link Digital
      // (la chiave condivisa è salvata in artifacts/{appId}/sharedDial/meta.digitalShareKey).
      allow create, update: if request.auth != null
        && request.resource.data.keys().hasAll(['updatedAt', 'cardsHTML', 'shiftLabel']);
      allow delete: if request.auth != null;
    }
