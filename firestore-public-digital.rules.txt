// Estratto da firestore.rules — publicDigital (proprietario ownerUid).
// Non incollare solo questo file: usa firestore.rules completo in Firebase Console.

    match /artifacts/{appId}/publicDigital/{shareId} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.ownerUid == request.auth.uid
        && request.resource.data.keys().hasAll(['ownerUid', 'updatedAt', 'cardsHTML']);
      allow update: if request.auth != null
        && resource.data.ownerUid == request.auth.uid
        && request.resource.data.ownerUid == request.auth.uid;
      allow delete: if request.auth != null
        && resource.data.ownerUid == request.auth.uid;
    }
