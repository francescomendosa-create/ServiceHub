// Incolla questo blocco dentro le regole Firestore del progetto servicehub-18309
// (Firebase Console → Firestore → Regole), dentro match /databases/{database}/documents { ... }

    match /artifacts/{appId}/publicDigital/{shareId} {
      // Chiunque con il link può LEGGERE lo snapshot (solo lettura sul tablet / PC remoto).
      allow read: if true;
      // Solo utenti autenticati possono creare/aggiornare documenti, e solo come proprietario.
      allow create: if request.auth != null
        && request.resource.data.ownerUid == request.auth.uid
        && request.resource.data.keys().hasAll(['ownerUid', 'updatedAt', 'cardsHTML']);
      allow update: if request.auth != null
        && resource.data.ownerUid == request.auth.uid
        && request.resource.data.ownerUid == request.auth.uid;
      allow delete: if request.auth != null
        && resource.data.ownerUid == request.auth.uid;
    }
